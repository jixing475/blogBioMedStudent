  mutate(High = factor(if_else(Sales <= 8, "No", "Yes")))

data <-

data <- data %>% 
  dplyr::mutate(
    High = as.factor(High),
    High = fct_collapse(High,
                        `0` = c("No"),
                        `1` = c("Yes")),
    High = fct_relevel(High,
                       "0",
                       "1"))
